class Person {
    /**class @property {string} name -identifica el nombre de la persona */
    name = "";
    email = ""; /**class @property {string} email -identifica el nombre de la persona */
    age = 0; /**class @property {number} age -identifica el nombre de la persona */
    resume = "";
    id=0; /**class @static {string} name -identifica el nombre de la persona */ /**class @static {number} id  -identifica el nombre de la persona */
    static total = 0;
    /**
     * 
     * @param {*} name 
     * @param {*} email 
     * @param {*} age 
     * @param {*} resume 
     */
    constructor(name, email, age, resume) {
        this.name = name;
        this.email = email.toLowerCase();
        this.age = (age < 18) ? 18 : age;
        this.resume = resume;
        Person.total++;
        this.id=Person.total;
    }
    static printTotal() {
        console.log(Person.total);
    }

    printInfo(div) {
        div.insertAdjacentHTML("beforeend", `
        <div id="card_${this.id}" class="card text-bg-primary mb-3" style="max-width: 18rem;">
        <div class="card-header">${this.name}</div>
        <div class="card-body">
            <h5 class="card-title">${this.email}</h5>
            <p class="card-text">${this.resume}.</p>
            <p class="card-text"><strong>${this.age}.</p>
        </div>
        </div>
        `)

    }
}

class Employee extends Person {
    department = "";
    salary = 0.0;
    constructor(name, email, age, resume, department, salary) {
        super(name, email, age, resume);
        this.department = department;
        this.salary = salary;
    }
    calculateSalary() {
        return ((this.salary * 30) * 1.16).toFixed(2);
    }
    printInfo(div) {
        super.printInfo(div);
        let cardBody=document.getElementById(`card_${this.id}`).getElementsByClassName("card-body")[0];
        cardBody.insertAdjacentHTML("beforeend",`
            <p class="card-text">${this.department}</p>
            <p class="card-text text-end"><strong>${this.calculateSalary()}</strong></p>
        `);
        // console.log(this.department, this.salary, this.calculateSalary());
    }
}
